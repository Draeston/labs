# Creating secrets manually
```
$ kubectl create secret generic jokes-secret --from-literal=funnyjoke=funnyjoke
secret/jokes-secret created

$ kubectl get secrets
NAME           TYPE     DATA   AGE
jokes-secret   Opaque   1      22s

$ kubectl get secret jokes-secret -o jsonpath='{.data}'
{"funnyjoke":"ZnVubnlqb2tl"}

$ echo "ZnVubnlqb2tl" | base64 --decode
funnyjoke
```

# Creating secrets with Helm

```
$ helm secrets install secret ./python-app -n default -f ./secrets.yaml

./secrets.yaml.dec is newer than ./secrets.yaml
NAME: secret
LAST DEPLOYED: Sun Feb 12 15:41:25 2023
NAMESPACE: default
STATUS: deployed
REVISION: 1
NOTES:
1. Get the application URL by running these commands:
  export POD_NAME=$(kubectl get pods --namespace default -l "app.kubernetes.io/name=python-app,app.kubernetes.io/instance=secret" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace default $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace default port-forward $POD_NAME 8080:$CONTAINER_PORT
```
